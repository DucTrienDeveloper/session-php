#include <iostream>
using namespace std;
int main()
{
    // khai bao bien
    int N;
    // Nhap du lieu
    cout << "Nhap 1 so nguyen duong N ";
    cin >> N;
    // tinh toan, xuat
    if (N < 100 || N>999)
        cout << "Du lieu khong dung ! ";
    else
    {
        int htr = N / 100;
        int hc = N / 10 % 10;
        int hdv = N % 10;
        switch (htr)
        {
        case 1:
            cout << "Mot tram ";
            break;
        case 2:
            cout << "Hai tram ";
            break;
        case 3:
            cout << "Ba tram ";
            break;
        case 4:
            cout << "Bon tram ";
            break;
        case 5:
            cout << "Nam tram ";
            break;
        case 6:
            cout << "Sau tram ";
            break;
        case 7:
            cout << "Bay tram ";
            break;
        case 8:
            cout << "Tam tram ";
            break;
        case 9:
            cout << "Chin tram ";
            break;
        }
        switch (hc)
        {
        case 1:
            cout << "Muoi ";
            break;
        case 2:
            cout << "Hai Muoi ";
            break;
        case 3:
            cout << "Ba Muoi ";
            break;
        case 4:
            cout << "Bon Muoi ";
            break;
        case 5:
            cout << "Nam Muoi ";
            break;
        case 6:
            cout << "Sau Muoi ";
            break;
        case 7:
            cout << "Bay Muoi ";
            break;
        case 8:
            cout << "Tam Muoi ";
            break;
        case 9:
            cout << "Chin Muoi ";
            break;
        }
        switch (hdv)
        {
        case 1:
            cout << "Mot ";
            break;
        case 2:
            cout << "Hai ";
            break;
        case 3:
            cout << "Ba ";
            break;
        case 4:
            cout << "Bon ";
            break;
        case 5:
            if (hc == 0)
                cout << "Nam";
            else
                cout << "Lam ";
            break;
        case 6:
            cout << "Sau ";
            break;
        case 7:
            cout << "Bay ";
            break;
        case 8:
            cout << "Tam ";
            break;
        case 9:
            cout << "Chin ";
            break;
        }
    }

    return 0;
}
